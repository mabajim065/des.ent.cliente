function fib(n) {
    // Caso base: fib(0) = 0, fib(1) = 1
    if (n <= 1) {
        return n;

    // Caso recursivo: fib(n) = fib(n-1) + fib(n-2)

function fib(n) {
    return n <= 1 ? n : fib(n - 1) + fib(n - 2);
